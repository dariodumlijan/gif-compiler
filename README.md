```sh
# -> Input folder
│
├── name_100x100_assets
│   │
│   ├── 1.jpg
│   └── 2.jpg
│
└── name_200x200_assets
    │
    ├── EN
    │   │
    │   ├── C01
    │   │   ├── 1.jpg
    │   │   └── 2.jpg
    │   │
    │   └── C02
    │       ├── 1.jpg
    │       └── 2.jpg
    │
    └── DE
        ├── C01
        │   ├── 1.jpg
        │   └── 2.jpg
        │
        └── C02
            ├── 1.jpg
```

## Project Setup

### Install

```bash
$ yarn
```

### Development

```bash
$ yarn dev
```

### Build

```bash
# For windows
$ yarn build:win

# For macOS
$ yarn build:mac

# For Linux
$ yarn build:linux
```
